<表达式> ::= <逻辑表达式> ('=' <空格>? <逻辑表达式>)*
<逻辑表达式> ::= <加法表达式> (<RelOp> <空格>? <加法表达式>)*
<RelOp> ::= '<' | '<=' | '>' | '>=' | '==' | '!='
<加法表达式> ::= <乘法表达式> ([+#x2D] <空格>? <乘法表达式>)*
<乘法表达式> ::= <因子> ([*/] <空格>? <因子>)*
<因子> ::=  (<数值> | <括号表达式> | <变量>) <空格>?
<括号表达式> ::= '(' <空格>? <表达式> ')'
<数值> ::= [0-9]+ ('l' | 'u' | 'll' | 'ull')?
<变量> ::= <标识符> ('(' <空格>? (<表达式> (',' <空格>? <表达式>)*)? ')')?
<关键字> ::= 'if'| 'while' | 'return'
<标识符> ::= [A-Za-z] [0-9A-Za-z_]* - <关键字>
<空格> ::= [#x9#xA#xD#x20]+
